// api/analyze.js export const config = { runtime: 'edge', // 关键：开启 Edge 模式，突破 10 秒限制 }; export default async function handler(req) { // Edge 模式下，必须检查请求方法 if (req.method !== 'POST') { return new Response(JSON.stringify({ error: 'Method Not Allowed' }), { status: 405, headers: { 'Content-Type': 'application/json' } }); } try { // 1. 读取前端发来的数据 const { birthDate, birthCity, mbti } = await req.json(); // 2. 检查 Key const apiKey = process.env.DEEPSEEK_API_KEY; if (!apiKey) { return new Response(JSON.stringify({ error: '服务器配置错误：未找到 Key' }), { status: 500, headers: { 'Content-Type': 'application/json' } }); } // 3. 向 DeepSeek 发送请求 const response = await fetch('https://api.deepseek.com/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${apiKey}` }, body: JSON.stringify({ model: 'deepseek-chat', messages: [ { role: "system", content: "你是一位精通八字与MBTI的专家。" }, { role: "user", content: `用户${birthDate}出生在${birthCity}，MBTI是${mbti}。请给出性格分析，300字以内。` } ], max_tokens: 800 // 限制字数以加快速度 }) }); // 4. 检查 DeepSeek 是否报错 if (!response.ok) { const errorText = await response.text(); return new Response(JSON.stringify({ error: `DeepSeek 报错: ${errorText}` }), { status: response.status, headers: { 'Content-Type': 'application/json' } }); } // 5. 返回结果 const data = await response.json(); return new Response(JSON.stringify(data), { status: 200, headers: { 'Content-Type': 'application/json' } }); } catch (error) { return new Response(JSON.stringify({ error: `服务器错误: ${error.message}` }), { status: 500, headers: { 'Content-Type': 'application/json' } }); } } // 强制刷新
